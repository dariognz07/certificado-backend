@app.post("/tutor/subir-certificado")
async def subir_certificado(
    usuario_id: int = Form(...),
    nombre_tutor: str = Form(...),
    email: str = Form(...),
    cert_p12: UploadFile = File(...),
    password_cert: str = Form(...),
    db: Session = Depends(database.get_db)
):
    contenido = await cert_p12.read()
    pass_enc = crypto.encriptar_password(password_cert)
    cert = models.Certificado(
        usuario_id=usuario_id,
        nombre_tutor=nombre_tutor,
        email=email,
        cert_p12=contenido,
        password_encriptada=pass_enc
    )
    db.add(cert)
    db.commit()
    return {"mensaje": "Certificado subido correctamente"}


@app.get("/tutor/certificados", response_model=schemas.ApiResponse)
def listar_certificados(estado: str = None, db: Session = Depends(database.get_db)):
    query = db.query(models.Certificado)
    if estado == "vigente":
        query = query.filter(models.Certificado.fecha_vencimiento >= func.now())
    elif estado == "vencido":
        query = query.filter(models.Certificado.fecha_vencimiento < func.now())
    return {"certificados": query.all()}

@app.post("/firmar-pdf")
async def firmar_pdf(usuario_id: int = Form(...), pdf: UploadFile = File(...), db: Session = Depends(database.get_db)):
    cert = db.query(models.Certificado).filter(models.Certificado.usuario_id == usuario_id).first()
    if not cert:
        raise HTTPException(status_code=404, detail="Certificado no encontrado")
    password = crypto.desencriptar_password(cert.password_encriptada)

from cryptography.fernet import Fernet
import os


FERNET_KEY = os.getenv("CERT_ENCRYPTION_KEY").encode()


def encriptar_password(password: str) -> str:
    f = Fernet(FERNET_KEY)
    return f.encrypt(password.encode()).decode()


def desencriptar_password(password_enc: str) -> str:
    f = Fernet(FERNET_KEY)
    return f.decrypt(password_enc.encode()).decode()


from pyhanko.sign import signers, timestamps
from pyhanko.pdf_utils.incremental_writer import IncrementalPdfFileWriter
from pyhanko.sign.fields import SigFieldSpec, append_signature_field
from io import BytesIO
from datetime import datetime


def firmar_pdf(pdf_bytes, cert_p12_bytes, password):
    signer = signers.SimpleSigner.load_pkcs12(cert_p12_bytes, passphrase=password.encode())
    out = BytesIO()
    w = IncrementalPdfFileWriter(BytesIO(pdf_bytes))


    if not w.root.get('/AcroForm'):
        append_signature_field(
            w,
            sig_field_spec=SigFieldSpec(
                sig_field_name='Signature1',
                box=(50, 50, 250, 100),
                on_page=0
            )
        )


    pdf_signer = signers.PdfSigner(
        signers.PdfSignatureMetadata(field_name='Signature1'),
        signer=signer,
        timestamper=timestamps.HTTPTimeStamper('http://timestamp.digicert.com')
    )


    pdf_signer.sign_pdf(w, output=out)
    return out.getvalue()

async def enviar_correo(destinatario: str, asunto: str, cuerpo: str):
    mensaje = MIMEMultipart()
    mensaje["From"] = MAIL_FROM
    mensaje["To"] = destinatario
    mensaje["Subject"] = asunto
    mensaje.attach(MIMEText(cuerpo, "plain", "utf-8"))


    await aiosmtplib.send(
        mensaje,
        hostname=MAIL_SERVER,
        port=MAIL_PORT,
        username=MAIL_USERNAME,
        password=MAIL_PASSWORD,
        start_tls=True,
    )

from sqlalchemy.orm import Session
from app import models, database, email_service
from datetime import datetime, timedelta
import asyncio


scheduler = BackgroundScheduler()


def chequear_y_notificar(db: Session):
    fecha_limite = datetime.now() + timedelta(days=30)


    certificados = db.query(models.Certificado).filter(
        models.Certificado.fecha_vencimiento <= fecha_limite,
        models.Certificado.fecha_vencimiento >= datetime.now()
    ).all()


    for cert in certificados:
        dias_restantes = (cert.fecha_vencimiento - datetime.now()).days


        ya_notificado = db.query(models.LogNotificacion).filter(
            models.LogNotificacion.usuario_id == cert.usuario_id,
            models.LogNotificacion.fecha_envio >= datetime.now().date()
        ).first()


        if not ya_notificado:
            asunto = f"Alerta: Certificado vence en {dias_restantes} días"
            cuerpo = f"""
Hola {cert.nombre_tutor},


Tu certificado digital vence el {cert.fecha_vencimiento.strftime('%d/%m/%Y')}.
Te quedan {dias_restantes} días para renovarlo.


Ingresa a la app para subir el nuevo certificado.
            """


            asyncio.run(email_service.enviar_correo(cert.email, asunto, cuerpo))


            log = models.LogNotificacion(
                usuario_id=cert.usuario_id,
                email=cert.email,
                mensaje=f"Notificación enviada - {dias_restantes} días restantes"
            )
            db.add(log)
            db.commit()


def iniciar_scheduler():
    scheduler.add_job(chequear_y_notificar, 'cron', hour=9, minute=0, args=[next(database.get_db())])
    scheduler.start()
